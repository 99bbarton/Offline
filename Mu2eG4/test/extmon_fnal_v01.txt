// detector location in room: the center is on the continuation
// of collimator2 center line.   Need to specify just one coord:

// Details of the detector.  All of these vectors should be
// of the same size: one entry per detector plane.
// The coordinates are relative to the detector box volume.

vector<double> extmon_fnal.sensor_zoffset = {-300., -280., -50., +0, +220, +300.};
vector<double> extmon_fnal.sensor_xoffset = {0., 0., 0., 0., 0., 0.};
