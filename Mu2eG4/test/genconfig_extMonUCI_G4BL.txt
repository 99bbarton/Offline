// This is on by default.  Turn it off.
bool conversionGun.do    = false;

// Pick any combination.
// If you enable fromG4BLFile you must NOT select 
// g4.physicsListName = "Minimal" in the beamline_geom.txt
bool primaryProtonGun.do = false;
bool particleGun.do      = false;
bool fromG4BLFile.do     = true;

// Primary proton gun.
// For the full set of parameters see EventGenerator/src/PrimaryProtonGun.cc

// cos(theta)=-1. perfectly aligned impact on production target.
double primaryProtonGun.czmin = -1.;
double primaryProtonGun.czmax = -1.;

// Displacement of PrimaryProtonGun relative to center of upstream face of the 
// the production target; in mm.
vector<double> beamDisplacementOnTarget = {0.,0.,0.};

// Beam spot sigma (in both transverse dimensions); in mm.
double primaryProtonGun.beamSpotSigma = 0.;   

// Range of times over which protons hit the target; in ns.
double primaryProtonGun.tmin  =    0.;
double primaryProtonGun.tmax  =  100.;

//
// --------------------------------------------------------------
// 

// Particle gun configured to shoot muons out of the target towards the TS.
// For the full set of parameters see EventGenerator/src/ParticleGun.cc

// PDG code for mu_minus
int particleGun.id = -2212;

// The middle of the production target; in Mu2e coordinates in mm.
//vector<double> particleGun.point = { 3904., 0., -6564.5 };

// 100 cm upstream of the middle of the production target.
// ( so that muons are not all absorbed in the target ).
//vector<double> particleGun.point = { 3904., 880., -11929.0 };
vector<double> particleGun.point = {3925.2, 0., -6079.5}; //{ 3924.64, 0., -6479.5 };

// Particles will be made uniformly in a box, centered on the above point.
// Half sides of the box are in mm.
vector<double> particleGun._halfLength = { 0., 0., 0. };

// For the MECO PS, particles more forward than -0.45 all go 
double         particleGun.czmin = 0.97029573; // 0.97175543; // 0.98358; // 0.97175;
double         particleGun.czmax = 0.97029573; // 0.97175543; // 0.98358; //0.97175;
double         particleGun.phimin = 0; //-0.1881;
double         particleGun.phimax = 0; //-0.1881;

// Momentum range.
double         particleGun.pmin  = 8888.6;
double         particleGun.pmax  = 8888.6;

bool particleGun.doHistograms = true;

//
// --------------------------------------------------------------
// 

// FromG4BLFile
// Read individual tracks

bool   fromG4BLFile.targetFrame  = false;
double fromG4BLFile.mean         = -1;
string fromG4BLFile.filename     = "particlelist.txt";
bool   fromG4BLFile.doHistograms = true;
vector<double> fromG4BLFile.g4beamlineOrigin = { 3904.0, 0.0, -7929.0 };
bool   fromG4BLFile.duplicateParticles = false;
