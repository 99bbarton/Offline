// The stay clear volume in which the whole tracker plus supports must fit.
// Tracker need not be centered in the stay clear volume.
// This volume is on the axis of the DS.
double ttracker.mother.rIn         =   380.; // mm
double ttracker.mother.rOut        =   840.; // mm
double ttracker.mother.halfLength  =  1635.; // mm
double ttracker.mother.z0          = 10175.; // mm

int    ttracker.layersPerSector             =    2;
int    ttracker.strawsPerManifold           =    1; //  1,10
double ttracker.deviceHalfSeparation        =   20.0;
double ttracker.supportHalfThickness        =   4.4;
// 96 straws.  All 3 of the following must be adjusted coherently
int    ttracker.manifoldsPerEnd             =   48; // 50, 5
double ttracker.strawGap             =     1.25;
vector<double> ttracker.manifoldHalfLengths = {3.125,5.0,5.21};
// 22 stations; NB, deviceSpacing is really station spacing.  Total length
// of ttracker is unchanged, spacing between stations is reduced
double ttracker.deviceSpacing        =   140.86;
int ttracker.numDevices = 44;
// rotation pattern: either 1, 2, or 3, see the code for descriptions
int    ttracker.rotationPattern    =    2;
// odd-numbered stations can be rotated WRT even to minimize dead stereo
// overlap. The optimal number depends on the pattern
double ttracker.oddStationRotation = 60.0;
