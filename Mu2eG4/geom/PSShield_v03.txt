// PS heat and radiation shield.  A two grooves version with parameters from Peter Kasper.
//
// Original author: Andrei Gaponenko

// Numbers from MARS times 10 (for cm=>mm) to define the shape
// The polycone
vector<double> PSShield.zPlane = {1050, 1700., 2290.5, 2290.5, 3340.5, 3340.5, 4580., 5000. };
vector<double> PSShield.rIn    = { 450., 300.,   300.,   250.,   250.,   300.,  300.,  150. };
vector<double> PSShield.rOut   = { 700., 700.,   700.,   700.,   700.,   700.,  700.,  700. };

// Each groove is a cylinder of the given dimensions, with the center
// at the given reference point and rotated by the given angles, which
// is subtracted from the polycone above.
int PSShield.nGrooves = 2;

// The groove for spent beam exit
double PSShield.groove1.r         = 200.; // mm
double PSShield.groove1.halfLengh = 2500.; // NB: == mars full length, but G4 ref pos is the center.
vector<double> PSShield.groove1.refPoint =  { -600, 0., -30.}; // MARS reference point for the groove
double PSShield.groove1.theta     = 14.00; // angle to Mu2e Z axis, degrees
double PSShield.groove1.phi       = 180.; // angle between Mu2e X axis and projection of the groove axis on XY, degrees.

// The groove ExtMonFNAL: just like the above but in a different direction
// Need to rotate the ref point (-600, 0) at 180 deg  ==> (cos(145*deg) * 600, sin(145*deg)*600) at 145 deg
double PSShield.groove2.r         = 200.; // mm
double PSShield.groove2.halfLengh = 2500.; // NB: == mars full length, but G4 ref pos is the center.
vector<double> PSShield.groove2.refPoint =  { -491.5, 344.1, -30.}; // MARS reference point for the groove
double PSShield.groove2.theta     = 14.00; // angle to Mu2e Z axis, degrees
double PSShield.groove2.phi       = 145.; // angle between Mu2e X axis and projection of the groove axis on XY, degrees.

// The shield is positioned on the PS axis.
// To fix its Z position we specify the offset w.r.t. to the production target center
// of the PSShield point at the most negative z.
//
// From Peter Kasper's e-mail May 17, 2012:
//     HRS downstream end z = -8018.0
//     Target center      z = -6164.5
double PSShield.zOffsetFromProductionTarget = -1853.5;

// FIXME: should be low density bronze
string PSShield.materialName = "G4_Cu";

bool PSShield.isVisible = true;
bool PSShield.isSolid = false;

int PSShield.verbosityLevel = 0;
